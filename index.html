const SHEET_NAME = "Marques"; const MSGS_SHEET = "Messages"; const PROFILE_SHEET = "Profil"; const NOTES_SHEET = "Notes"; const OBJ_SHEET = "Objectifs"; function doGet(e) { const callback = e.parameter.callback; const data = getAllData(); if(callback) { return ContentService.createTextOutput(callback + '(' + JSON.stringify(data) + ')') .setMimeType(ContentService.MimeType.JAVASCRIPT); } return ContentService.createTextOutput(JSON.stringify(data)) .setMimeType(ContentService.MimeType.JSON); } function doPost(e) { try { const data = JSON.parse(e.postData.contents); const action = data.action; let result; if (action === "getAll") result = getAllData(); else if (action === "saveMarques") result = saveMarques(data.marques); else if (action === "saveMsgs") result = saveMsgs(data.msgs); else if (action === "saveProfile") result = saveProfile(data.profile); else if (action === "saveNotes") result = saveNotes(data.notes); else if (action === "saveObjectifs") result = saveObjectifs(data.objectifs); else result = { error: "Action inconnue" }; return ContentService.createTextOutput(JSON.stringify(result)) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService.createTextOutput(JSON.stringify({ error: err.toString() })) .setMimeType(ContentService.MimeType.JSON); } } function getSheet(name) { const ss = SpreadsheetApp.getActiveSpreadsheet(); let sheet = ss.getSheetByName(name); if (!sheet) sheet = ss.insertSheet(name); return sheet; } function saveMarques(marques) { const sheet = getSheet(SHEET_NAME); sheet.clearContents(); const headers = ["id","nom","categorie","contact_nom","canal_principal","email","instagram","site","telephone", "statut","date_contact","date_r1","date_r2","date_relance","resultat","type_contenu","statut_contenu", "date_livraison","produit_recu","date_produit","tarif_propose","montant","paiement","note", "paie_delais","recontacter","tags","notes","date_ajout"]; sheet.appendRow(headers); marques.forEach(m => sheet.appendRow(headers.map(h => m[h] || ""))); return { success: true, count: marques.length }; } function getMarques() { const sheet = getSheet(SHEET_NAME); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; const headers = data[0]; return data.slice(1).map(row => { const obj = {}; headers.forEach((h, i) => { obj[h] = row[i] !== undefined ? String(row[i]) : ""; }); return obj; }).filter(m => m.nom); } function saveMsgs(msgs) { const sheet = getSheet(MSGS_SHEET); sheet.clearContents(); sheet.appendRow(["id", "titre", "corps"]); msgs.forEach(m => sheet.appendRow([m.id||"", m.titre||"", m.corps||""])); return { success: true }; } function getMsgs() { const sheet = getSheet(MSGS_SHEET); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; return data.slice(1).map(row => ({ id: row[0], titre: row[1], corps: row[2] })).filter(m => m.titre); } function saveProfile(profile) { const sheet = getSheet(PROFILE_SHEET); sheet.clearContents(); sheet.appendRow(["nom", "spec", "bio", "insta", "photo"]); sheet.appendRow([profile.nom||"", profile.spec||"", profile.bio||"", profile.insta||"", profile.photo||""]); return { success: true }; } function getProfile() { const sheet = getSheet(PROFILE_SHEET); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return {}; const headers = data[0]; const row = data[1]; const obj = {}; headers.forEach((h, i) => { obj[h] = row[i] || ""; }); return obj; } function saveNotes(notes) { const sheet = getSheet(NOTES_SHEET); sheet.clearContents(); sheet.appendRow(["note"]); notes.forEach(n => sheet.appendRow([n])); return { success: true }; } function getNotes() { const sheet = getSheet(NOTES_SHEET); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return []; return data.slice(1).map(row => row[0]).filter(n => n); } function saveObjectifs(obj) { const sheet = getSheet(OBJ_SHEET); sheet.clearContents(); sheet.appendRow(["revenus", "collabs"]); sheet.appendRow([obj.revenus||0, obj.collabs||0]); return { success: true }; } function getObjectifs() { const sheet = getSheet(OBJ_SHEET); const data = sheet.getDataRange().getValues(); if (data.length <= 1) return { revenus: 0, collabs: 0 }; return { revenus: Number(data[1][0])||0, collabs: Number(data[1][1])||0 }; } function getAllData() { return { marques: getMarques(), msgs: getMsgs(), profile: getProfile(), notes: getNotes(), objectifs: getObjectifs() }; }
